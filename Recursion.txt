Definition :
Recursion is a programming technique where a function calls itself to solve smaller instances of the same problem until a base condition is reached.
Without a base case, recursion will continue forever and cause Stack Overflow Error.
Recursion breaks a big problem into smaller sub-problems of the same type.

01. Print 10 to 1!
.
.
public class Main {
    public static void main(String[] args) {
        Print(10);
    }

    static int Print(int n) {
        if (n < 1) {
            return 1;
        }
        System.out.println(n);
        return Print(n - 1);
    }
}

    static int Print(int n) {

>>>> Binary Search Using Recursion!
.
.
public class Main {
    public static void main(String[] args) {
        int[] arr = {32,21,4,344,3,1,5};
        int target = 344;
        System.out.println(Search(arr,target,0,arr.length-1));
    }

    static int Search(int[] arr, int target, int start, int end) {
        if(start > end ) return -1;

        int mid = start + (end - start) / 2;

        if(arr[mid] == target) {
            return mid;
        }

        if(arr[mid] > target) {
            return Search(arr, target, start, mid - 1);
        }

        // if(arr[mid] < target) 
        return Search(arr, target, mid + 1, end);
    }
}
